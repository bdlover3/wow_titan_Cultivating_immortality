### 2.2 初始化事件

```lua
-- 方案一：PLAYER_LOGIN（推荐，登录完成后触发，SavedVariables 已可用）
local frame = CreateFrame("Frame")
frame:RegisterEvent("PLAYER_LOGIN")
frame:SetScript("OnEvent", function(self, event, ...)
    -- 初始化代码
end)

-- 方案二：ADDON_LOADED（在加载阶段触发，比 PLAYER_LOGIN 早）
local frame = CreateFrame("Frame")
frame:RegisterEvent("ADDON_LOADED")
frame:SetScript("OnEvent", function(self, event, addonName)
    if addonName ~= "MyAddon" then return end
    self:UnregisterEvent("ADDON_LOADED")
end)

-- 方案三：PLAYER_ENTERING_WORLD（每次进入世界都触发）
local frame = CreateFrame("Frame")
frame:RegisterEvent("PLAYER_ENTERING_WORLD")
frame:SetScript("OnEvent", function(self, event, isInitialLogin, isReloadingUi)
end)
```

### 10.9 PlayerModel（3D模型）

```lua
local model = CreateFrame("PlayerModel", nil, parent)
model:SetSize(200, 200)
model:SetDisplayInfo(displayID)    -- 设置显示模型（通过DisplayID）
model:SetCreature(creatureID)      -- 设置NPC模型（通过NPC ID，需NPC数据已缓存）
model:SetUnit("unit")              -- 设置单位模型（如 "player"）
model:SetCamera(cameraIndex)       -- 设置摄像机
model:SetPosition(x, y, z)         -- 设置模型位置偏移
model:SetFacing(radians)           -- 设置朝向
model:SetRotation(radians)         -- 设置旋转
model:SetModelScale(scale)         -- 设置模型缩放
model:SetCamDistanceScale(scale)    -- 设置摄像机距离缩放
model:SetPortraitZoom(zoom)        -- 设置头像缩放
model:ClearModel()                 -- 清除当前模型
model:GetModel()                   -- 获取当前模型路径（成功返回路径字符串，失败返回nil/空字符串）
```

**注意**:
- `SetCreature` 需要NPC数据已被客户端缓存，首次使用可能返回空模型
- `SetDisplayInfo` 直接使用DisplayID加载，不依赖NPC缓存
- 推荐使用 `pcall` 包裹所有模型操作，防止无效ID导致报错
- `GetModel()` 返回模型文件路径（如 "character\\bloodelf\\female\\bloodelffemale.m2"），可用于验证模型是否加载成功

## 十五、设置面板 API（3.80.1 新 Settings API）

```lua
-- 注册配置面板（垂直列表风格，推荐）
local category, layout = Settings.RegisterVerticalLayoutCategory("MyAddon")

-- 方式一：Proxy Setting（需自行管理 SavedVariables）
if MyAddonSettings == nil then
    MyAddonSettings = {}
end
if MyAddonSettings.enabled == nil then
    MyAddonSettings.enabled = true
end

local function GetEnabled()
    return MyAddonSettings.enabled
end
local function SetEnabled(value)
    MyAddonSettings.enabled = value
end

local enabledSetting = Settings.RegisterProxySetting(
    category,                    -- 所属分类
    "MY_ADDON_ENABLED",          -- 变量名（唯一标识）
    Settings.VarType.Boolean,    -- 变量类型
    "启用功能",                   -- 显示名称
    true,                        -- 默认值
    GetEnabled,                  -- 获取值函数
    SetEnabled                   -- 设置值函数
)
Settings.CreateCheckbox(category, enabledSetting)

-- 方式二：AddOn Setting（自动管理 SavedVariables）
local alphaSetting = Settings.RegisterAddOnSetting(
    category,                    -- 所属分类
    "MY_ADDON_ALPHA",            -- 变量名
    "alpha",                     -- SavedVariables 表中的键名
    MyAddonSettings,             -- SavedVariables 表
    Settings.VarType.Number,     -- 变量类型
    "透明度",                     -- 显示名称
    1.0                          -- 默认值
)
alphaSetting:SetValueChangedCallback(function(setting, value)
    -- 值变化时的回调
end)

local options = Settings.CreateSliderOptions(0, 1, 0.01)
Settings.CreateSlider(category, alphaSetting, options, "透明度设置")

-- 注册子分类
local subcategory, subcatLayout = Settings.RegisterVerticalLayoutSubcategory(category, "高级设置")

-- 注册完成
Settings.RegisterAddOnCategory(category)

-- 打开设置面板到指定分类
Settings.OpenToCategory(category:GetID())
```
